🚀 Tugas Mandiri: Rekening Bank (Bonus 2 & 3)

Sesuai instruksi slide 14, berikut adalah implementasi kelas RekeningBank dengan validasi saldo tidak boleh negatif.


class RekeningBank {
  String _nomorRekening;
  double _saldo;

  RekeningBank(this._nomorRekening, this._saldo);

  // Getter untuk melihat saldo
  double get saldo => _saldo;

  // Method Deposit
  void deposit(double jumlah) {
    if (jumlah > 0) {
      _saldo += jumlah;
      print('Berhasil deposit: Rp$jumlah. Saldo sekarang: Rp$_saldo');
    }
  }

  // Method Tarik dengan Validasi Saldo
  void tarik(double jumlah) {
    if (jumlah > 0 && _saldo >= jumlah) {
      _saldo -= jumlah;
      print('Berhasil tarik: Rp$jumlah. Sisa saldo: Rp$_saldo');
    } else {
      print('Gagal: Saldo tidak mencukupi atau jumlah tidak valid!');
    }
  }

  void cekSaldo() {
    print('No. Rekening: $_nomorRekening | Saldo: Rp$_saldo');
  }
}

void main() {
  var akun = RekeningBank("123-456", 500000);
  akun.cekSaldo();
  akun.deposit(200000);
  akun.tarik(100000);
  akun.tarik(1000000); // Uji coba validasi saldo kurang
}
