}

🏗️ Jenis Constructor di Dart

  • Default Constructor: ClassName()
  • Named Constructor: ClassName.name()
  • Parameterized Constructor: ClassName(this.param)
  • Constant Constructor: const ClassName() (semua field final, immutable, hemat memori)
  • Factory Constructor: (lanjutan)

📍 Named Constructor – Contoh Mobil

import 'dart:convert';

class Mobil {
  String? merk;
  int? tahun;

  Mobil(this.merk, this.tahun);               // parameterized
  Mobil.baru() { merk = 'Toyota'; tahun = 2023; } // named
  Mobil.dariJson(Map<String, dynamic> json) {
    merk = json['merk'];
    tahun = json['tahun'];
  }

  @override
  String toString() => 'Mobil{merk: $merk, tahun: $tahun}';
}

🧊 Constant Constructor – Warna

class Warna {
  final int r, g, b;
  const Warna(this.r, this.g, this.b);
  @override String toString() => 'Warna(r:$r, g:$g, b:$b)';
}

void main() {
  const merah = Warna(255,0,0);
  const merahLagi = Warna(255,0,0); // instance yang sama
  print(merah);
  print(identical(merah, merahLagi)); // true
}

⚙️ Initializer List – Titik

class Titik {
  final double x, y;
  Titik(double x, double y)
      : x = x.abs(), y = y.abs() {  // inisialisasi sebelum body
    print('Titik dibuat di ($x, $y)');
  }
}

🛡️ Enkapsulasi

Konsep: menyembunyikan detail implementasi. Tujuan: keamanan, kontrol akses, fleksibilitas.
Di Dart: gunakan _ (underscore) untuk private member (berlaku per file).

🔒 Public vs Private – Person

class Person {
  String _name;   // private
  int age;        // public
  Person(this._name, this.age);
}

📥 Getter – membaca private

class Person {
  String _name;
  int age;
  Person(this._name, this.age);

  String get getName1 => _name;       // getter style 1
  String getName() => _name;           // method biasa
}

📤 Setter – modifikasi dengan validasi

class Person {
  String _name;
  int age;
  Person(this._name, this.age);

  set setName1(String name) => _name = name;
  void setName(String name) { _name = name; }
}

✅ Produk – Getter/Setter dengan validasi

class Produk {
  String _nama;
  double _harga;
  Produk(this._nama, this._harga);

  String get nama => _nama;
  double get harga => _harga;

  set nama(String value) {
    if (value.length >= 3) _nama = value;
    else print('Nama minimal 3 karakter');
  }

  set harga(double value) {
    if (value > 0) _harga = value;
    else print('Harga harus > 0');
  }
}

🔐 Studi Kasus: User – Login & ganti password

class User {
  String _username;
  String _password;
  User(this._username, this._password);

  String get username => _username;
  bool login(String username, String password) =>
      _username == username && _password == password;

  void gantiPassword(String lama, String baru) {
    if (_password != lama) {
      print('Password lama salah');
    } else if (baru.length < 8) {
      print('Password baru minimal 8 karakter');
    } else {
      _password = baru;
      print('Password berhasil diubah');
    }
  }
}

🧪 Penyelesaian Soal Latihan 1 & 2 (RekeningBank)

Soal 1: Buat kelas RekeningBank dengan field private _nomorRekening, _saldo. Method: deposit(), tarik(), cekSaldo(). Validasi saldo tidak boleh negatif.

Soal 2: Buat program sederhana simulasi transaksi.

🏦 Kelas RekeningBank (lengkap)

class RekeningBank {
  String _nomorRekening;
  double _saldo;

  RekeningBank(this._nomorRekening, this._saldo) {
    if (_saldo < 0) _saldo = 0; // pastikan awal tidak negatif
  }

  // Getter (opsional)
  String get nomorRekening => _nomorRekening;
  double get saldo => _saldo;

  void deposit(double jumlah) {
    if (jumlah > 0) {
      _saldo += jumlah;
      print('✔️ Deposit Rp$jumlah berhasil. Saldo sekarang: Rp$_saldo');
    } else {
      print('❌ Jumlah deposit harus positif.');
    }
  }

  void tarik(double jumlah) {
    if (jumlah <= 0) {
      print('❌ Jumlah tarik harus positif.');
    } else if (jumlah > _saldo) {
      print('❌ Saldo tidak mencukupi. Saldo: Rp$_saldo');
    } else {
      _saldo -= jumlah;
      print('✔️ Tarik Rp$jumlah berhasil. Saldo sekarang: Rp$_saldo');
    }
  }

  void cekSaldo() {
    print('💰 Saldo rekening $_nomorRekening : Rp$_saldo');
  }
}

💻 Simulasi Transaksi (Soal 2)

void main() {
  print('========== SIMULASI TRANSAKSI BANK ==========');

  var rek1 = RekeningBank('001-234567-89', 500000);
  var rek2 = RekeningBank('002-987654-32', 250000);

  print('\n--- Data Awal ---');
  rek1.cekSaldo();
  rek2.cekSaldo();

  print('\n--- Transaksi Rekening 1 ---');
  rek1.deposit(200000);
  rek1.tarik(100000);
  rek1.tarik(700000);  // gagal (saldo kurang)
  rek1.deposit(-50000); // gagal (negatif)
  rek1.cekSaldo();

  print('\n--- Transaksi Rekening 2 ---');
  rek2.tarik(50000);
  rek2.tarik(300000);  // gagal
  rek2.deposit(1000000);
  rek2.cekSaldo();

  print('\n--- Status Akhir ---');
  rek1.cekSaldo();
  rek2.cekSaldo();

  print('\n========== SIMULASI SELESAI ==========');
}
