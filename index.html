Membuat kelas RekeningBank dengan field _saldo dan method tarik() yang memvalidasi sisa saldo tidak boleh negatif.[span_31](end_span)


// Kode Tugas RekeningBank Anda
class RekeningBank {
  String _nomorRekening;
  double _saldo;

  RekeningBank(this._nomorRekening, this._saldo);

  void deposit(double jml) { if (jml > 0) _saldo += jml; }
  
  void tarik(double jml) {
    if (jml > 0 && _saldo >= jml) {
      _saldo -= jml;
      print('Tarik berhasil.');
    } else {
      print('Saldo tidak cukup.');
    }
  }

  void cekSaldo() => print('Saldo: $_saldo');
}
